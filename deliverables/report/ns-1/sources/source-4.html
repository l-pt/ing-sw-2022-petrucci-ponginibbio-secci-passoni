418     private void handleCharacter() {

420             Player player = view.getPlayerFromName(name);
421             int coins = player.getCoins();
422             List<Character> usableCharacters = new ArrayList<>(view.getCharacters().stream().filter(c -> coins >= c.getCost()).toList());
423             int tableTotalStudents = player.getSchool().getTables().values().stream().mapToInt(List::size).sum();
424             if (tableTotalStudents == 0) {
425                 //If the table is empty, we can't play Character10, so remove it from usableCharacters
426                 usableCharacters.removeIf(c -> c instanceof Character10);
427             } else if (tableTotalStudents == 10 * PawnColor.values().length) {
428                 //If the table is full, we can't play Character11
429                 usableCharacters.removeIf(c -> c instanceof Character11);
430             }
431             //Remove Character10 from the list if it is not possible to make a legal move
432             Set<PawnColor> fullTableColors = Arrays.stream(PawnColor.values()).filter(color -> player.getSchool().getTableCount(color) == 10).collect(Collectors.toSet());
433             if (!fullTableColors.isEmpty() && fullTableColors.containsAll(player.getSchool().getEntrance().stream().map(Student::getColor).toList())) {
434                 usableCharacters.removeIf(c -> c instanceof Character10);
435             }
436             if (view.isExpert() && !usableCharacters.isEmpty()) {

438                 titleLbl.setAlignmentX(Component.CENTER_ALIGNMENT);
